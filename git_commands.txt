============================================================
GIT BASIC COMMANDS REFERENCE
============================================================

# 1. SETUP AND INITIALIZATION

# Initialize a new Git repository in the current folder.
git init

# Download an existing repository from a remote URL (like GitHub).
git clone [URL]

# Set your author name for all future commits (do this once).
git config --global user.name "Your Name"

# Set your email address for all future commits (do this once).
git config --global user.email "your@email.com"

------------------------------------------------------------
# 2. THE CORE WORKFLOW (TRACKING CHANGES)

# View the status of your working directory, staged files, and branch.
git status

# Add a specific file to the Staging Area (prepare for commit).
git add [filename]

# Add ALL modified and untracked files to the Staging Area.
git add .

# Record the snapshot of the Staging Area permanently.
# The -m flag lets you write a short, descriptive message.
git commit -m "Descriptive commit message"

# Remove a file from the Staging Area without deleting it from the disk.
git reset [filename]

------------------------------------------------------------
# 3. VIEWING HISTORY AND DIFFS

# Show the entire commit history (author, date, hash, message).
git log

# Show a concise, one-line summary of the commit history.
git log --oneline

# Show unstaged changes: the difference between the Working Directory and Staging Area.
git diff

# Show staged changes: the difference between the Staging Area and the last commit.
git diff --staged

------------------------------------------------------------
# 4. BRANCHING AND MERGING

# List all local branches.
git branch

# Create a new branch (but do NOT switch to it).
git branch [new-branch-name]

# Switch to an existing branch.
git checkout [branch-name]

# Shortcut: Create a new branch AND switch to it immediately.
git checkout -b [new-branch-name]

# Merge changes from the specified branch into your CURRENT branch.
git merge [branch-name]

------------------------------------------------------------
# 5. REMOTE INTERACTION (e.g., GitHub)

# Push (upload) your local commits to the remote repository.
# (Use -u once for a new branch, then 'git push' works automatically)
git push -u origin [branch-name]

# Pull (download and merge) changes from the remote repository.
git pull origin [branch-name]

# Download changes from the remote but DO NOT merge them immediately.
git fetch
